\begin{question}
	f(0,0) = 0
	
	f(a,b) = a, para b = 1
	
	f(a,b) = a + f(a,b -1), para b $\geq$ 1
   \begin{enumerate}
   		\item \textbf{base:} (0,1) $\in$ f
   		\item \textbf{passo recursivo:} se b $\geq$ 1 e (a,b-1) $\in$ f 
   		\item \textbf{fechamento:} só pertence a N o número que pode ser obtido de acordo
   		com (1) e (2)
   \end{enumerate}
\end{question}
