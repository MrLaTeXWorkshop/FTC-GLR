\begin{question}
	(A $\cap$ B)' = A' $\cup$ B'

	\begin{equation}
	  \begin{split}
	  		A = & [1,3,5,7] \\
			B = & [2,5,7,8,9] \\
			C = & [1,2,3,5,7,8,9] \Rightarrow \text{Conjunto universo}\\
		\end{split}
	\end{equation}
	
	\begin{equation}
		\begin{aligned}
			D = & A \cap B = [5,7] \\
			(A \cap B)' = & C - D = [1,2,3,8,9] 
		\end{aligned}
	\end{equation}

	\begin{equation}
		\begin{aligned}
			A' = & C - A = [2,8,9] \\
			B' = & C - B = [1,3] \\
		\end{aligned}
	\end{equation}
	
	\begin{equation}
		\begin{aligned}
			A' \cup B' = &  [1,2,3,8,9]
		\end{aligned}
	\end{equation}
	
	\begin{equation}
		\begin{aligned}
			A' \cup B' = &  (A \cap B)'
		\end{aligned}
	\end{equation}
	
\end{question}
