\begin{question}
	Prove que $1 + 2^{n} < 3^{n}$ para $n < 2$
   \begin{equation}
	  \begin{split}
	  		1 + 2^{n} &< 3^{n} \\
			2^{0} + 2^{n} &< 3^{n} \\
			log(2)^{0} + log(2)^{n} &< log(3)^{n} \\
			0*log(2) + n*log(2) &< n* log(3)\\
			log(2) &< log(3) \\
		\end{split}
	\end{equation}
\end{question}
